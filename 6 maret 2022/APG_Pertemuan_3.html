Q1 <- quantile(data1$Fuel.x1., .25)
Q3 <- quantile(data1$Fuel.x1., .75)
IQR <- (Q3 - Q1)

upper_bound <- Q3 + (IQR * 1.5)
lower_bound <- Q1 - (IQR * 1.5)

eliminated <- subset(data1, data1$Fuel.x1. > lower_bound & data1$Fuel.x1. < upper_bound)
